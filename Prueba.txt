import pyaudio
import wave

def grabar_audio(nombre_archivo="grabacion.wav", duracion=5):
    """
    Graba audio y lo guarda en formato WAV
    
    Args:
        nombre_archivo: Nombre del archivo de salida (por defecto: grabacion.wav)
        duracion: Duración de la grabación en segundos (por defecto: 5)
    """
    # Configuración de parámetros
    CHUNK = 1024  # Tamaño del buffer
    FORMAT = pyaudio.paInt16  # Formato de audio (16 bits)
    CHANNELS = 2  # Estéreo (usa 1 para mono)
    RATE = 44100  # Frecuencia de muestreo (44.1 kHz)
    
    # Inicializar PyAudio
    audio = pyaudio.PyAudio()
    
    # Abrir stream de audio
    stream = audio.open(format=FORMAT,
                       channels=CHANNELS,
                       rate=RATE,
                       input=True,
                       frames_per_buffer=CHUNK)
    
    print(f"Grabando durante {duracion} segundos...")
    
    frames = []
    
    # Grabar audio
    for i in range(0, int(RATE / CHUNK * duracion)):
        data = stream.read(CHUNK)
        frames.append(data)
    
    print("Grabación finalizada.")
    
    # Detener y cerrar el stream
    stream.stop_stream()
    stream.close()
    audio.terminate()
    
    # Guardar el audio en archivo WAV
    with wave.open(nombre_archivo, 'wb') as wf:
        wf.setnchannels(CHANNELS)
        wf.setsampwidth(audio.get_sample_size(FORMAT))
        wf.setframerate(RATE)
        wf.writeframes(b''.join(frames))
    
    print(f"Audio guardado en: {nombre_archivo}")

if __name__ == "__main__":
    # Ejemplo de uso
    grabar_audio("mi_grabacion.wav", duracion=5)
