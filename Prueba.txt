import pdfplumber
import re

def extraer_texto_pdf(ruta_pdf):
    texto = ""
    with pdfplumber.open(ruta_pdf) as pdf:
        for pagina in pdf.pages:
            pagina_texto = pagina.extract_text()
            if pagina_texto:
                texto += pagina_texto + " "
    return texto


def buscar_palabras_contexto(
    texto,
    palabra_base,
    palabra_objetivo,
    rango=15
):
    resultados = []

    # Normalizar texto
    texto = texto.lower()

    # Extraer palabras (elimina signos)
    palabras = re.findall(r'\b\w+\b', texto)

    for i, palabra in enumerate(palabras):
        if palabra == palabra_base.lower():

            inicio = max(0, i - rango)
            fin = min(len(palabras), i + rango + 1)

            contexto = palabras[inicio:fin]

            if palabra_objetivo.lower() in contexto:
                resultados.append({
                    "posicion_palabra_base": i,
                    "rango": (inicio, fin),
                    "contexto": " ".join(contexto)
                })

    return resultados


if __name__ == "__main__":
    ruta_pdf = "archivo.pdf"
    palabra_base = "residencia"
    palabra_objetivo = "clave"
    rango = 15

    texto_pdf = extraer_texto_pdf(ruta_pdf)

    coincidencias = buscar_palabras_contexto(
        texto_pdf,
        palabra_base,
        palabra_objetivo,
        rango
    )

    if coincidencias:
        for idx, c in enumerate(coincidencias, 1):
            print(f"\n📍 Coincidencia {idx}")
            print(f"Posición de '{palabra_base}': {c['posicion_palabra_base']}")
            print(f"Rango de palabras: {c['rango']}")
            print(f"Contexto:\n{c['contexto']}")
    else:
        print("❌ No se encontraron coincidencias.")
